public class Main {
    public static void main(String[] args) {
        int number = 10; // You can change this to any integer

        // Check if the number is even
        if (number % 2 == 0) {
            System.out.println(number + " is an even number.");
        } else {
            System.out.println(number + " is an odd number.");
        }
    }
}
